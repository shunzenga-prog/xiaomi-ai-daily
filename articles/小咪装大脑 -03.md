## 一、环境准备

在开始之前，我们需要准备好以下环境：

### 1.1 Node.js 环境

OpenMemory 的客户端库主要使用 Node.js，建议版本 18+：

```bash
# 检查当前版本
node -v

# 如果版本太低，使用 nvm 升级
nvm install 20
nvm use 20

# 验证安装
node -v  # 应该输出 v20.x.x
npm -v   # 应该输出 9.x 或更高
```

### 1.2 Python 环境（可选）

如果你需要用 Python 集成，准备 3.8+：

```bash
# 检查 Python 版本
python3 --version

# 安装虚拟环境工具
pip3 install virtualenv

# 创建项目虚拟环境
python3 -m virtualenv venv
source venv/bin/activate
```

### 1.3 数据库准备

OpenMemory 支持多种存储后端：

- **SQLite**：适合开发和小型项目（默认）
- **PostgreSQL**：适合生产环境
- **Redis**：适合缓存层

我们先用 SQLite 快速上手：

```bash
# 创建项目目录
mkdir openmemory-demo && cd openmemory-demo

# 初始化项目
npm init -y

# 安装 OpenMemory 客户端
npm install @openmemory/client
```

## 三、集成代码逐行讲解

### 3.1 基础客户端初始化

```javascript
// memory-client.js
const { OpenMemoryClient } = require('@openmemory/client');

// 初始化客户端
const client = new OpenMemoryClient({
  baseUrl: 'http://localhost:3000',
  apiKey: 'your-secret-api-key-change-this',
  // 可选：重试配置
  retryConfig: {
    maxRetries: 3,
    retryDelay: 1000,
  },
});

module.exports = client;
```

**逐行解析：**
- 第 1 行：导入官方客户端库
- 第 4-11 行：创建客户端实例，配置服务端地址和认证密钥
- 第 13 行：导出供其他模块使用

### 3.2 存储记忆

```javascript
// save-memory.js
const client = require('./memory-client');

async function saveMemory(userId, content, metadata = {}) {
  try {
    const memory = await client.memories.create({
      userId,           // 用户标识
      content,          // 记忆内容
      metadata: {
        ...metadata,
        createdAt: new Date().toISOString(),
        source: 'chat', // 来源：chat/note/file 等
      },
      tags: metadata.tags || [], // 标签用于分类检索
    });

    console.log('✅ 记忆已保存:', memory.id);
    return memory;
  } catch (error) {
    console.error('❌ 保存失败:', error.message);
    throw error;
  }
}

// 使用示例
saveMemory('user-123', '用户喜欢喝拿铁咖啡', {
  category: 'preference',
  tags: ['饮品', '偏好'],
});
```

**关键点：**
- `userId`：用于隔离不同用户的记忆
- `content`：记忆的核心内容，支持纯文本或结构化 JSON
- `metadata`：扩展信息，可以存储时间、来源、置信度等
- `tags`：标签系统，方便后续检索

### 3.3 检索记忆

```javascript
// search-memory.js
const client = require('./memory-client');

async function searchMemories(userId, query, options = {}) {
  const results = await client.memories.search({
    userId,
    query,              // 搜索关键词
    limit: options.limit || 10,
    filters: {
      tags: options.tags,      // 按标签过滤
      dateFrom: options.from,  // 时间范围
      dateTo: options.to,
      categories: options.categories,
    },
    // 相似度阈值（0-1）
    minScore: options.minScore || 0.7,
  });

  return results.memories;
}

// 使用示例
async function getUserPreferences(userId) {
  const memories = await searchMemories(userId, '喜欢', {
    tags: ['偏好'],
    limit: 5,
  });

  return memories.map(m => m.content);
}
```

### 3.4 更新和删除记忆

```javascript
// update-memory.js
const client = require('./memory-client');

// 更新记忆
async function updateMemory(memoryId, updates) {
  return client.memories.update(memoryId, {
    content: updates.content,
    metadata: updates.metadata,
    tags: updates.tags,
  });
}

// 删除记忆
async function deleteMemory(memoryId) {
  await client.memories.delete(memoryId);
  console.log('🗑️ 记忆已删除');
}

// 批量删除（比如用户要求"忘记所有关于咖啡的事"）
async function deleteMemoriesByTag(userId, tag) {
  const memories = await client.memories.list({ userId, tags: [tag] });
  
  for (const memory of memories.memories) {
    await client.memories.delete(memory.id);
  }
  
  console.log(`🗑️ 已删除 ${memories.memories.length} 条记忆`);
}
```

### 3.5 在对话中集成

```javascript
// chat-with-memory.js
const client = require('./memory-client');

class MemoryAwareChat {
  constructor(userId) {
    this.userId = userId;
  }

  // 对话前：加载相关记忆
  async loadContext(message) {
    const memories = await client.memories.search({
      userId: this.userId,
      query: message,
      limit: 5,
    });

    // 构建系统提示
    const context = memories.memories
      .map(m => `- ${m.content}`)
      .join('\n');

    return context;
  }

  // 对话后：保存新记忆
  async saveInsight(message, response, extractedMemories) {
    for (const memory of extractedMemories) {
      await client.memories.create({
        userId: this.userId,
        content: memory,
        metadata: {
          relatedMessage: message,
          relatedResponse: response,
        },
      });
    }
  }
}

// 使用示例
const chat = new MemoryAwareChat('user-123');

async function handleMessage(userMessage) {
  // 1. 加载上下文
  const context = await chat.loadContext(userMessage);
  
  // 2. 调用 LLM（带上记忆上下文）
  const response = await callLLM(userMessage, context);
  
  // 3. 提取并保存新记忆
  const newMemories = extractMemories(userMessage, response);
  await chat.saveInsight(userMessage, response, newMemories);
  
  return response;
}
```

## 四、常见问题和解决方案

### 4.1 连接失败

**问题：** `ECONNREFUSED` 或超时

**解决方案：**
```bash
# 1. 检查服务是否运行
curl http://localhost:3000/health

# 2. 检查防火墙
sudo ufw allow 3000/tcp

# 3. 检查端口占用
lsof -i :3000

# 4. 重启服务
npm restart
```

### 4.2 认证失败

**问题：** `401 Unauthorized`

**解决方案：**
- 检查 `.env` 中的 `API_KEY` 是否正确
- 确保请求头格式正确：`Authorization: Bearer <your-key>`
- 不要硬编码密钥，使用环境变量

### 4.3 记忆检索不准确

**问题：** 搜索结果不相关

**解决方案：**
```javascript
// 调整搜索策略
const results = await client.memories.search({
  query,
  // 1. 提高相似度阈值
  minScore: 0.85,
  // 2. 添加标签过滤
  filters: { tags: ['重要'] },
  // 3. 限制时间范围
  filters: { dateFrom: '2024-01-01' },
  // 4. 增加返回数量，人工筛选
  limit: 20,
});
```

### 4.4 数据库性能问题

**问题：** 查询变慢，记忆多了之后卡顿

**解决方案：**
```bash
# 1. 从 SQLite 迁移到 PostgreSQL
DATABASE_URL=postgresql://user:pass@localhost:5432/openmemory

# 2. 添加索引
psql -d openmemory -c "CREATE INDEX idx_user_tags ON memories(user_id, tags);"

# 3. 定期清理旧记忆
# 添加定时任务，删除 90 天前的非重要记忆
```

### 4.5 记忆冲突

**问题：** 同一用户有矛盾的记忆

**解决方案：**
```javascript
// 实现记忆去重和合并
async function deduplicateMemories(userId, newMemory) {
  const existing = await client.memories.search({
    userId,
    query: newMemory.content,
    minScore: 0.9, // 高相似度
  });

  if (existing.memories.length > 0) {
    // 更新现有记忆而不是创建新的
    await client.memories.update(existing.memories[0].id, {
      content: newMemory.content,
      metadata: {
        ...existing.memories[0].metadata,
        updatedAt: new Date().toISOString(),
        version: (existing.memories[0].metadata.version || 1) + 1,
      },
    });
    return 'updated';
  }

  // 创建新记忆
  await client.memories.create(newMemory);
  return 'created';
}
```

## 六、部署到生产环境

### 6.1 使用 Docker 部署

创建 `Dockerfile`：

```dockerfile
FROM node:20-alpine

WORKDIR /app

COPY package*.json ./
RUN npm ci --only=production

COPY . .

EXPOSE 3000

CMD ["npm", "start"]
```

构建和运行：

```bash
# 构建镜像
docker build -t openmemory-server .

# 运行容器
docker run -d \
  -p 3000:3000 \
  -v openmemory-data:/app/data \
  -e API_KEY=$(openssl rand -hex 32) \
  --name openmemory \
  openmemory-server
```

### 6.2 使用 Docker Compose

创建 `docker-compose.yml`：

```yaml
version: '3.8'

services:
  openmemory:
    build: .
    ports:
      - "3000:3000"
    environment:
      - DATABASE_URL=postgresql://postgres:password@db:5432/openmemory
      - API_KEY=${API_KEY}
    depends_on:
      - db
    volumes:
      - openmemory-data:/app/data

  db:
    image: postgres:15
    environment:
      - POSTGRES_PASSWORD=password
      - POSTGRES_DB=openmemory
    volumes:
      - postgres-data:/var/lib/postgresql/data

volumes:
  openmemory-data:
  postgres-data:
```

启动：

```bash
# 生成 API 密钥
export API_KEY=$(openssl rand -hex 32)

# 启动服务
docker-compose up -d

# 查看日志
docker-compose logs -f
```
